		1, -- duration
		Enum.EasingStyle.Sine,
		Enum.EasingDirection.Out
	)

	local total = #frameNames
	for i, name in ipairs(frameNames) do
		local frame = parent:WaitForChild(name)
	
		-- Set AnchorPoint to center of frame

		frame.Position = UDim2.new(1.2, 0, 0.5, 0)

		}

		end)
	end
